1.  #include  <bits/stdc++.h>
2.  using  namespace std;
3.  #define  int  long  long
4.  int A[6], B[100005], C[100005][6];
5.  int n;
6.  int minsum(vector<pair<int,  int>>  &V)
7.  {
8.  if  (V.empty())
9.  return  0;
10.  sort(V.begin(), V.end());
11.  vector<pair<int,  int>> U{V[V.size()  -  1]};
12.  for  (int i = V.size()  -  2; i >=  0;  --i)
13.  if  (V[i].second > U[U.size()  -  1].second)
14.  U.push_back(V[i]);
15.  int mi = min(U[0].first, U[U.size()  -  1].second);
16.  for  (int i =  0; i < U.size()  -  1;  ++i)
17.  mi = min(mi, U[i].second + U[i +  1].first);
18.  return mi;
19.  }
20.  int func(int v)
21.  {
22.  vector<pair<int,  int>> V, U;
23.  for  (int i =  1; i < n;  ++i)
24.  {
25.  int l =  1e18, r =  1e18;
26.  for  (int j =  0; j <  6;  ++j)
27.  {
28.  if  (C[i][j]  <= v)
29.  l = min(l, v - C[i][j]);
30.  if  (C[i][j]  >= v)
31.  r = min(r, C[i][j]  - v);
32.  }
33.  V.emplace_back(l, r);
34.  }
35.  return minsum(V);
36.  }
